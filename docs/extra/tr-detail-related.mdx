## İlişkili ürünler detay sayfası
Ürün detay sayfası ilişkili ürün yapısı için kullanılabilir.



Bir tane <code>related</code> isminde alt şablon oluşturup kodlamayı yapınca detay sayfasında çağırabiliriz. 
<code>related</code> klasöründe <code>default.tpl</code> şablonu  olarak kodlanmışsa şu şekilde çağırabiliriz.



## Default (Resimli)
```html title="smarty default"
  {if $P.RELATED_PRODUCTS_IDS1_COUNT }
    <div class="related-wrapper" data-pid="{$P.ID}">
        {include file='block/product-detail/tpl/related/default.tpl'}
    </div>
  {/if}
```
```html title="smarty related default"
<div id="related-product" class="row w-100 my-1" v-cloak>
    <div class="col-12 mb-1 text-body fw-bold feature-title" v-if="RELATEDS.length">{#color_selection#|default:'Diğer Seçenekler'}</div>
    <div v-for="(P, index) in RELATEDS" :key="index" class="col-3 col-sm-2 mb-1" :class="{ 'order-1':  !getSelected(P.URL) } ">
        <a :href="P.URL" class="w-100 image-wrapper border border-2x border-round" :class="{ 'border-primary':  getSelected(P.URL) }">
            <div class="image-inner">
                <figure class="image-zoom">
                    <img :src="P.IMAGE.SMALL" :alt="P.TITLE">
                </figure>
            </div>
        </a>
    </div>
</div>
```
```javascript title="javascript"

    const relatedProduct = {
    data() {
        return {
            endpoints: {
                related: `/srv/service/product/get-related-products/${PRODUCT_DATA[0].id}/1`
            },
            RELATEDS: []
        }
    },
    methods: {
        load() {
            axios.get(this.endpoints.related)
                .then((res) => {
                    this.RELATEDS = res.data.PRODUCTS
                    // console.log(this.RELATEDS)
                })
                .catch(function (error) {
                    console.warn(error);
                })
        },
        getSelected(url) {
            return window.location.pathname.replace("/", "") == url
        },
    },
    mounted() {
        this.getSelected()
        this.load()
    },
}
Vue.createApp(relatedProduct).mount('#related-product');
```


## Select option 

```html title="smarty related select option"
<div id="related-product" class="w-100 my-1" v-cloak>
    <div class="w-100 mb-1 text-body fw-bold feature-title" v-if="RELATEDS.length">{#color_selection#|default:'Diğer Seçenekler'} Renk seçimi</div>
    <select name="related" class="related-select form-control form-control-md mb-1" onchange="window.location.href=this.value">
    {literal}
        <option :value="P.URL" class="col-12" v-for="(P, index) in RELATEDS">{{P.TITLE}}</option>
    {/literal}
    </select>
</div>
```
```javascript title="javascript"
    const relatedProduct = {
    data() {
        return {
            endpoints: {
                related: `/srv/service/product/get-related-products/${PRODUCT_DATA[0].id}/1`
            },
            RELATEDS: []
        }
    },
    methods: {
        load() {
            axios.get(this.endpoints.related)
                .then((res) => {
                    this.RELATEDS = [...res.data.PRODUCTS]
                })
                .catch(function (error) {
                    console.warn(error);
                })
        }
    },
    mounted() {
        this.load()
    },
}
Vue.createApp(relatedProduct).mount('#related-product');
```
