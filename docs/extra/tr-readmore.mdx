## Daha fazla oku Üst Alt Vitrin
```html title="Smarty"
{category id="{$TABLE_ID}" assign="CATEGORY"}
{if $CATEGORY|@count ||  $SHOWCASE_CONTENT}
<div class="w-100 readmore-wrapper position-relative pb-2">
   <div class="readmore-content ease overflow-hidden p-1" style="height: 80px;">
         {if $CATEGORY.NAME}
            <h2 class="category-name">{$CATEGORY.NAME}</h2>
         {/if}
         {if $CATEGORY.DESCRIPTION}
            <div class="category-description">{$CATEGORY.DESCRIPTION}</div>
         {/if}
         {if $SHOWCASE_CONTENT}
            <div class="showcase-content">{$SHOWCASE_CONTENT}</div>
         {/if}
   </div>
   <div class="readmore-btn-wrapper w-100 position-absolute bottom-0">
      <span class="p-1 more readmore-btn d-block py-1 text-underline fw-bold">{#read_more#}</span>
      <span class="p-1 less readmore-btn d-block py-1 text-underline fw-bold d-none">{#read_less#}</span>
   </div>
</div>
{/if}
```

## Javascript
```javascript title="javascript"
let content = T(`.readmore-content`);
let contentImg=T(`.readmore-content img`); // resim olduğu durumlarda

let contentHeight = parseInt(content.height()) + parseInt(contentImg.height()); // contentImg alanı resim olan durumlar için
let contentScrollHeight = content[0].scrollHeight;
if (content && contentHeight >= contentScrollHeight) {
  T(`.readmore-btn-wrapper`)[0].remove();
} else {
  content.height(contentHeight);
  T(`.more`).attr("data-height", contentScrollHeight);
  T(`.less`).attr("data-height", contentHeight);
}
T(`.readmore-btn`).on(`click`, function (e) {
  T(`.readmore-btn`).toggleClass("d-none");
  content.toggleClass("active");
  content.height(T(this).attr("data-height"));
});
```
